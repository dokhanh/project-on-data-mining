\end{frame}

\begin{frame}
    \frametitle{Contexte de l'étude et objectif de l'analyse}
    \begin{itemize}
        \item $25$ promotions de $1962$ à $2005$.
        \item Éléments pris en compte: Origines (nationalités, modes d'admission), parcours (diplômes obtenus), activités professionnelles (types d'entreprises, responsabilités, salaires).
        \item Objectifs: découvrir des relations de ces variables, éventuellement des évolutions au cours du temps.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Descriptions des données utilisées}
    \begin{itemize}
        \item L'enquête \textsl{Fonctions et Salaires Édition 2009}.
        \item La plupart des variables sont catégorielles.
        \item Les variables utilisées: \textsl{diplôme}, \textsl{promotion}, \textsl{admission}, \textsl{origine}, \textsl{type d'entreprise}, \textsl{taille d'entreprise}, \textsl{responsabilité}, \textsl{activité professionnelle}, \textsl{création d'entreprise}, \textsl{salaire brut}.
        \item Diviser les variables en deux thèmes: parcours/origine et création d'entreprise/domaine professionnel.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Description des méthodes utilisées}
    \begin{itemize}
        \item On ne traite que les variables catégorielles par ACM.
        \item Regrouper les modalités et les valeurs des variables \textsl{promotion} et \textsl{salaire brut} en $4$ groupes, chaque groupe $\approx$ $25\%$ des individus.
    \end{itemize}

\begin{frame}
	\frametitle{Conclusion}
	
	\begin{itemize}
		\item{Corrélation étroite entre les nationalités, les modes d'admission et les diplômes obtenus par les élèves après l'école}
		\item{Pour chaque domaine d'activité, on a des entreprises de taille typique, des postes typiques occupés par les élèves, et les niveaux de salaire correspondants.}
		\item{Opposition privé/publique et start-up/non start-up}
		\item{Voies d'admission et les parcours ont évolué au cours des ces $25$ promotions}
	\end{itemize}
